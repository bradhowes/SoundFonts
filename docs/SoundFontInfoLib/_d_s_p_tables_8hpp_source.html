56 struct SineLookup {
57  inline constexpr static size_t TableSize = 4096;
58 
64  inline static Float sine(Float radians) {
65  if (radians < 0.0) return -sin(-radians);
66  while (radians > TwoPI) radians -= TwoPI;
67  if (radians <= HalfPI) return interpolate(radians);
68  if (radians <= PI) return interpolate(PI - radians);
69  if (radians <= 3 * HalfPI) return -interpolate(radians - PI);
70  return -interpolate(TwoPI - radians);
71  }
72 
73 private:
74  inline constexpr static Float TableScale = (TableSize - 1) / HalfPI;
75  inline constexpr static Float Scaling = HalfPI / (TableSize - 1);
76 
77  inline static Float interpolate(Float radians) {
78  Float phase = std::clamp(radians, 0.0, HalfPI) * TableScale;
79  int index = int(phase);
80  Float partial = phase - index;
81  Float value = lookup_[index] * (1.0 - partial);
82  if (partial > 0.0) value += lookup_[index + 1] * partial;
83  return value;
84  }
85 
86  static Float value(size_t index) { return std::sin(index * Scaling); }
87

89  SineLookup() = delete;
